//Fibonacci series in java

public class Main
{
	public static void main(String[] args) {
	    int num1 = 0, num2 =1, num3, temp;
	    System.out.println(num1 +" "+ num2);
	    for(int i = 0; i <=5; i++){
	        num3 = num1 + num2;
	        System.out.println(num3);
	        temp = num2;
	        num2 =num3;
	        num1 =temp;
	    }
		
	}
}
